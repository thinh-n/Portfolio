#create for loop for the set of numbers to test
for number in range(1,16):
									
	#if the number is divisible by 15, then print fizz buzz
	if float(number/15) == int(number/15):
		print("Fizz Buzz")
										
	#if the number is divisible by 5, then print buzz
	elif float(number/5) == int(number/5):
		print("Buzz")
											
	#if the number is divisible by 3, then print fizz
	elif float(number/3) == int(number/3):
		print("Fizz")
											
	#all other numbers will be printed normally
	else:
		print(number)
